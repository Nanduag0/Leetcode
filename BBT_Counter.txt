class Solution {
  public:
    long long int countBT(int h) 
    {
        long long int dp[h+1];
        memset(dp,0,sizeof(dp));
        dp[0]=1;
        dp[1]=1;
        for(int i=2;i<=h;i++)
        {
            dp[i]=(dp[i-1]*(((2*dp[i-2])%1000000007)+dp[i-1])%1000000007)%1000000007;
            //cout<<dp[i]<<" ";
        }
        return dp[h]%1000000007;
        
        
    }
};
